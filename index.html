L = 0

def convert(speed):
R = int(-20 * speed + 1600)
L = int(20 * speed + 1400)
return(R,L)
#Default direction
def front(input):
RPL.servoWrite(R, convert(input)[0])
RPL.servoWrite(L, convert(input)[1])
def left(input):
RPL.servoWrite(R,convert(input)[0])
def right(input):
RPL.servoWrite(L,convert(input)[1])

One with the actual code, one to run it, and one to test it for errors.
